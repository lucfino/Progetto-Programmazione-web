1 /**
  2  * dato l'id dell'oggetto e il tipo di fade che si vuole effettuare, applica il fade al parent
  3  * dell'oggetto passato
  4  * @param {string} idOggetto L'id dell'oggetto figlio dell'oggetto a cui si vuole appliare il fade
  5  * @param {mixed} tipo Il tipo di fade: 1 se fade-in, qualunque altro valore altrimenti 
  6  */
  7 function effettoScomparsa(idOggetto,tipo){
  8     var cont;
  9     document.getElementById(idOggetto).parentNode.setAttribute('fadeIn', 0);
 10     document.getElementById(idOggetto).parentNode.setAttribute('fadeOut', 1);
 11     if(tipo == '1'){
 12         cont = "document.getElementById(\""+idOggetto+"\").parentNode";
 13         fadeEffectIn(cont);
 14     }
 15     else{
 16         cont = "document.getElementById(\""+idOggetto+"\").parentNode";
 17         fadeEffectOut(cont);
 18     }
 19 }

 21 /**
 22  * Simula un effetto di fade-in dato il parametro
 23  * @param {string} cont Una stringa del tipo document.getElementById('esempio')
 24  */
 25 function fadeEffectIn(cont){
 26     var errore = eval(cont);
 27     if(parseFloat(errore.attributes['fadeIn'].nodeValue) < 1){
 28         errore.style.opacity = parseFloat(errore.style.opacity)+0.1;
 29         if(parseFloat(errore.style.opacity)>1.1){
 30             errore.style.opacity = '0';
 31             errore.attributes['fadeIn'].nodeValue = 1;
 32             }
 33         errore.attributes['fadeIn'].nodeValue = parseFloat(errore.attributes['fadeIn'].nodeValue)+0.1;
 34         setTimeout("fadeEffectIn('"+cont+"')", 100);
 35         }
 36     else
 37         return;
 38 }

 40 /**
 41  * Simula un effetto di fade-out dato il parametro
 42  * @param {string} cont Una stringa del tipo document.getElementById('esempio')
 43  */
 44 function fadeEffectOut(cont){
 45     var errore = eval(cont);
 46     if(parseFloat(errore.attributes['fadeOut'].nodeValue) > 0){
 47         errore.style.opacity = parseFloat(errore.style.opacity)-0.1;
 48         errore.attributes['fadeOut'].nodeValue = parseFloat(errore.attributes['fadeOut'].nodeValue)-0.1;
 49         if(parseFloat(errore.style.opacity)<0.1){
 50             errore.style.opacity = '0';
 51             errore.attributes['fadeOut'].nodeValue = 0;}
 52         setTimeout("fadeEffectOut('"+cont+"')", 100);}
 53     else
 54         return;
 55 }

 62 /**
 63  * La funzione, dato un evento, mostra vicino al mouse una
 64  * foto grazie alla stringa k
 65  * @param {event} event L'oggetto evento
 66  * @param {string} k nome della foto da visualizzare
 67   */
 68 function mostraFoto(event,path){
 69     var q = document.getElementById('foto');
 70     if(q!=undefined)
 71         q.parentNode.removeChild(q);
 72     var x = event.pageX;
 73     var y = event.pageY;
 74     var foto;
 75     var evento = document.getElementById('eventi');
 76     foto = document.createElement('img');
 77     foto.style.position = 'absolute';
 78     foto.style.top = (y+12)+'px';
 79     foto.style.left = (x+12)+'px';
 80     foto.style.border = '3px solid black';
 81     foto.style.width = '200px';
 82     foto.style.height = '200px';
 83     foto.style.backgroundColor = '#ffff88';
 84     foto.setAttribute('id', 'foto');
 85     foto.style.padding = '0px';
 86     foto.src = path;
 87     evento.appendChild(foto);
 88 }

 98 /**
 99  * Compone la stringa da inviare al server contenente i valori della form
100  * @param {string} formname il nome della form da inviare
101  * @author presa da internet
102  */
103 function getquerystring(formname) {
104     var form = document.forms[formname];
105 	var qstr = "";
106 
107     /**
108       * Compone una stringa nel formato x-www-form-urlencoded
109       * @param {string} name
110       * @param {string} value
111       * @author presa da internet
112       */
113     function GetElemValue(name, value) {
114         qstr += (qstr.length > 0 ? "&" : "")
115             + escape(name).replace(/\+/g, "%2B") + "="
116             + escape(value ? value : "").replace(/\+/g, "%2B");
117 			//+ escape(value ? value : "").replace(/\n/g, "%0D");
118     }
119 	var elemArray = form.elements;
120     for (var i = 0; i < elemArray.length; i++) {
121         var element = elemArray[i];
122         var elemType = element.type.toUpperCase();
123         var elemName = element.name;
124         if (elemName) {
125             if (elemType == "TEXT"
126                     || elemType == "TEXTAREA"
127                     || elemType == "PASSWORD"
128 					|| elemType == "BUTTON"
129 					|| elemType == "RESET"
130 					|| elemType == "SUBMIT"
131 					|| elemType == "FILE"
132 					|| elemType == "IMAGE"
133                     || elemType == "HIDDEN")
134                 GetElemValue(elemName, element.value);
135             else if (elemType == "CHECKBOX" && element.checked)
136                 GetElemValue(elemName,
137                     element.value ? element.value : "On");
138             else if (elemType == "RADIO" && element.checked)
139                 GetElemValue(elemName, element.value);
140             else if (elemType.indexOf("SELECT") != -1)
141                 for (var j = 0; j < element.options.length; j++) {
142                     var option = element.options[j];
143                     if (option.selected)
144                         GetElemValue(elemName,
145                             option.value ? option.value : option.text);
146                 }
147         }
148     }
149     qstr+=qstr+'&aj=1';
150     return qstr;
151 }
